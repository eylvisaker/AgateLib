struct VS_OUTPUT
{
	float4 hposition : POSITION;
	float2 texcoord0 : TEXCOORD0;
};

struct PS_OUTPUT
{
	float4 color : COLOR;
};

sampler testTexture;

PS_OUTPUT main( VS_OUTPUT IN )
{
	PS_OUTPUT OUT;

	OUT.color = tex2D( testTexture, IN.texcoord0 ); // Add texel color to vertex color

	return OUT;
}
